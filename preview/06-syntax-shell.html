Midnightbash 
#!/usr/bin/env bash
# Vivid Life Theme — install

set -euo pipefail
FLAVOR="${1:-midnight}"

for f in midnight twilight dawn noon; do
    cp "themes/${f}.json" "$HOME/.config/"
done
echo "✓ installed ${FLAVOR}"
